# Comprehensive GWAS-Style Data Mining Pipeline

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import statsmodels.api as sm
from statsmodels.discrete.discrete_model import Logit
from sklearn.decomposition import PCA
from sklearn.manifold import TSNE
from sklearn.cluster import KMeans
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, roc_curve, auc

# For reproducibility
np.random.seed(42)

#############################
# 1. Data Simulation
#############################

# Settings for simulation
n_individuals = 100  # number of individuals
n_snps = 20          # number of SNPs

# Simulate genotype data: values 0, 1, or 2, assuming an allele frequency of 0.3 (Hardy-Weinberg equilibrium)
allele_freq = 0.3
genotypes = np.random.binomial(2, allele_freq, size=(n_individuals, n_snps))
snp_names = [f'SNP_{i+1}' for i in range(n_snps)]
df_geno = pd.DataFrame(genotypes, columns=snp_names)

# Simulate a binary phenotype.
# Let one SNP (say, SNP_5) affect the phenotype
intercept = -1.0  
beta = 1.5  # effect size for SNP_5
log_odds = intercept + beta * df_geno['SNP_5']
probabilities = 1 / (1 + np.exp(-log_odds))
phenotype = np.random.binomial(1, probabilities, n_individuals)
df_pheno = pd.DataFrame({'Phenotype': phenotype})

# Combine genotype and phenotype
df = pd.concat([df_geno, df_pheno], axis=1)

print("Simulated Data (first 5 rows):")
print(df.head())

#############################
# 2. Exploratory Data Analysis (EDA)
#############################

# Summary statistics
print("\nGenotype Summary Statistics:")
print(df_geno.describe())

# Plot distribution of one SNP as an example
plt.figure(figsize=(6,4))
sns.countplot(x='SNP_5', data=df_geno)
plt.title("Distribution of SNP_5 Genotypes")
plt.xlabel("Genotype")
plt.ylabel("Count")
plt.show()

# Phenotype distribution
plt.figure(figsize=(6,4))
sns.countplot(x='Phenotype', data=df_pheno)
plt.title("Phenotype Distribution")
plt.xlabel("Phenotype")
plt.ylabel("Count")
plt.show()

#############################
# 3. Dimensionality Reduction
#############################

# PCA on genotype data
pca = PCA(n_components=2)
pca_result = pca.fit_transform(df_geno)
df['PCA1'] = pca_result[:, 0]
df['PCA2'] = pca_result[:, 1]

plt.figure(figsize=(6,6))
sns.scatterplot(x='PCA1', y='PCA2', hue='Phenotype', data=df, palette='Set1')
plt.title("PCA of Genotype Data")
plt.show()

# t-SNE on genotype data
tsne = TSNE(n_components=2, random_state=42, perplexity=10)
tsne_result = tsne.fit_transform(df_geno)
df['tSNE1'] = tsne_result[:, 0]
df['tSNE2'] = tsne_result[:, 1]

plt.figure(figsize=(6,6))
sns.scatterplot(x='tSNE1', y='tSNE2', hue='Phenotype', data=df, palette='Set2')
plt.title("t-SNE of Genotype Data")
plt.show()

#############################
# 4. Clustering Analysis
#############################

# K-Means clustering on the PCA components
kmeans = KMeans(n_clusters=2, random_state=42)
clusters = kmeans.fit_predict(pca_result)
df['Cluster'] = clusters

plt.figure(figsize=(6,6))
sns.scatterplot(x='PCA1', y='PCA2', hue='Cluster', data=df, palette='Set1', style=df['Phenotype'])
plt.title("K-Means Clustering on PCA")
plt.show()

#############################
# 5. Association Testing
#############################

p_values = []
for snp in snp_names:
    X = df[[snp]]
    X = sm.add_constant(X)
    y = df['Phenotype']
    
    try:
        model = Logit(y, X)
        result = model.fit(disp=0)
        p = result.pvalues[snp]
    except Exception as e:
        p = np.nan
        print(f"Error in {snp}: {e}")
    p_values.append(p)

df_results = pd.DataFrame({
    'SNP': snp_names,
    'p_value': p_values
})
df_results['-log10(p)'] = -np.log10(df_results['p_value'])
print("\nAssociation Test Results:")
print(df_results)

# Manhattan Plot
plt.figure(figsize=(10, 6))
plt.scatter(range(1, n_snps+1), df_results['-log10(p)'], color='blue')
plt.xlabel('SNP Index')
plt.ylabel('-log10(p-value)')
plt.title('Manhattan Plot (Simulated GWAS)')
# Bonferroni correction threshold
bonf_threshold = -np.log10(0.05 / n_snps)
plt.axhline(bonf_threshold, color='red', linestyle='--', label='Bonferroni threshold')
plt.legend()
plt.show()

# Q-Q Plot
observed = np.sort(df_results['p_value'].dropna())
expected = np.linspace(0, 1, len(observed)+2)[1:-1]
plt.figure(figsize=(6, 6))
plt.scatter(-np.log10(expected), -np.log10(observed), color='purple')
plt.plot([-np.log10(expected[0]), -np.log10(expected[-1])],
         [-np.log10(expected[0]), -np.log10(expected[-1])],
         color='black', linestyle='--')
plt.xlabel('Expected -log10(p)')
plt.ylabel('Observed -log10(p)')
plt.title('Q-Q Plot of p-values')
plt.show()

#############################
# 6. Feature Selection and Classification
#############################

# Use Random Forest to predict phenotype from genotype data
rf = RandomForestClassifier(n_estimators=100, random_state=42)
X_rf = df_geno.values
y_rf = df['Phenotype'].values
rf.fit(X_rf, y_rf)
y_pred = rf.predict(X_rf)
accuracy = accuracy_score(y_rf, y_pred)
print("\nRandom Forest Classification Accuracy on Training Data: {:.2f}%".format(accuracy*100))

# Feature importances
importances = rf.feature_importances_
feature_importances = pd.DataFrame({
    'SNP': snp_names,
    'Importance': importances
}).sort_values(by='Importance', ascending=False)
print("\nFeature Importances:")
print(feature_importances)

plt.figure(figsize=(8,4))
sns.barplot(x='SNP', y='Importance', data=feature_importances, palette='viridis')
plt.title("Random Forest Feature Importances")
plt.xticks(rotation=45)
plt.show()

#############################
# 7. Exploratory Mining (Correlation Analysis)
#############################

# Calculate correlation between SNPs and phenotype
corrs = {}
for snp in snp_names:
    corrs[snp] = df[[snp, 'Phenotype']].corr().iloc[0,1]
df_corr = pd.DataFrame(list(corrs.items()), columns=['SNP', 'Correlation'])
print("\nSNP-Phenotype Correlations:")
print(df_corr)

plt.figure(figsize=(8,4))
sns.barplot(x='SNP', y='Correlation', data=df_corr, palette='coolwarm')
plt.title("Correlation of Each SNP with Phenotype")
plt.xticks(rotation=45)
plt.show()
